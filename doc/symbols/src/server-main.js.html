  6 /** for the 'express' library */
  7 var express = require('express');
  8 var app = express();
  9 /** nodejs crypto module */
 10 var crypto = require('crypto');
 11 /**
 12 tell express to use the 'client' folder as a source of static files such as
 13 index.html, index.css, etc...
 14 */
 15 app.use(express.static('client'));
 16 /** create nodejs server*/
 17 var server = require('http').Server(app);
 18 /** for the 'socket.io' library*/
 19 var io = require('socket.io')(server);
 20 /** filesystem*/
 21 var fs = require('fs');
 22 /** the sql database interface*/
 23 var cmmsql = require('./cmmsql.js');
 24 /** create new object for the sql database*/
 25 var db = new cmmsql('cmm.db');

 27 
 28 /** port number that the server listens on*/
 29 var portNum = 3000;
 30 /** list of active session tokens and their associated usernames*/
 31 /** usage: activeUsers[token] = username;*/
 32 var activeUsers = {};
 33 var numActiveUsers = 0;
 34 /** list of connected socket IDs, their authentication status, client IP address,
 35 and associated username (if authenticated)
 36 usage: activeSockets[socketId].authenticated = true/false;
 37 usage: activeSockets[socketId].clientIp = ipAddress;
 38 usage: activeSockets[socketId].username = username;
 39 */
 40 var activeSockets = {};
 41 var numActiveSockets = 0;
 42 
 43 
 44 /** this is called when a socket is connected
 45 io.on('connection', function(socket) {
 46     var socketId = socket.id;
 47     var clientIp = socket.request.connection.remoteAddress;
 48 
 49     //add socket to list of connected sockets
 50     activeSockets[socketId] = {};
 51     activeSockets[socketId].authenticated = false;
 52     activeSockets[socketId].clientIp = clientIp;
 53     activeSockets[socketId].username = null;
 54 
 55     numActiveSockets++;
 56     console.log('connected: ' + clientIp);
 57     console.log('total connected sockets: ' + numActiveSockets);
 58 
 59     //register event function callbacks
 60     registerEventFuncs(socket, socketId, clientIp);
 61 });
 62 
 63 
 64 /** start the server
 65 server.listen(portNum, function() {
 66     console.log('server listening on port ' + portNum);
 67 });

 71 /**
 72 container to hold the socket event functions which are registered as callbacks
 73 when a socket is connected
 74 */
 75 function registerEventFuncs(socket, socketId, clientIp) {
 76     /** sent by client to request that a new user account be created */
 77     socket.on('account create attempt', function(userInfo) {
 78         console.log('request to create new account with details:\n' +
 79             'user: ' + userInfo.username + ' pass: ' + userInfo.password);
 80 
 81 
 82         /** send request to database to create the new account */
 83         db.adduser(userInfo.username, userInfo.password, function(err, result) {
 84             if (err) {
 85                 //there was a problem creating the new account
 86                 //notify the client
 87                 console.log('account creation failed');
 88                 socket.emit('account create fail');
 89             } else {
 90                 contact = 'success';
 91                 //new account created successfully in database
 92                 //notify the client
 93                 console.log('account successfully created');
 94                 socket.emit('account create success');
 95             }
 96         });
 97     });
 98 
 99     /** authentication event (NOT to be confused with login event) */
100     socket.on('auth attempt', function(token) {
101         for (var key in activeUsers) {
102             //if the token represents a currently logged-in user...
103             if (key == token) {
104                 //mark the socket as authenticated and associate a username with it
105                 activeSockets[socketId].authenticated = true;
106                 activeSockets[socketId].username = activeUsers[token];
107                 console.log('socket: ' + socketId + ' authenticated by token: ' +
108                     token + ' for username: ' + activeUsers[token]);
109                 break;
110             }
111 
112         }
113     });
114 
115     /** called when a client submits a chat message to the server */
116     socket.on('chat submit', function(msgData) {
117         //user who is authenticated on this socket is the sender of the message
118         var chatSenderUsername = activeSockets[socketId].username;
119 
120         //TESTING
121         console.log(chatSenderUsername + ' submitted to chat id: ' + msgData.chatRoomId +
122             ' \nthis message: ' + msgData.msg);
123         console.log('rec ' + msgData.receivers);
124 
125         //send the message to the receiving users if they are logged in
126         msgData.receivers.push(chatSenderUsername);
127         for (var i in msgData.receivers) {
128             for (var j in activeSockets) {
129                 if (msgData.receivers[i] === activeSockets[j].username) {
130                     io.to(j).emit('chat deliver', { chatRoomId: msgData.chatRoomId, sender: chatSenderUsername, msg: msgData.msg });
131 
132                     break;
133                 }
134             }
135         }
136         //log the message to the database
137         db.logroom(msgData.chatRoomId, chatSenderUsername, msgData.msg, function(err, result) {
138             if (err) {
139                 console.log('error logging message to database with chatRoomId: ' +
140                     msgData.chatRoomId);
141             } else {
142                 //message logged to database successfully
143             }
144         });
145     });

208     /** login event (NOT to be confused with authentication event) */
209     socket.on('login attempt', function(userInfo) {
210         console.log('user attempting to login...\n' +
211             'user: ' + userInfo.username + ' pass: ' + userInfo.password);
212 
213         //query the database to check the user's account details
214         db.listusers('mainroom', function(err, usersInDb) {
215             //make sure that an account with the username exists
216             var usernameExists = false;
217             for (var key in usersInDb) {
218                 if (userInfo.username === usersInDb[key].user) {
219                     usernameExists = true;
220                     break;
221                 }
222             }
223             //username exists, so now we need to verify password
224             if (usernameExists) {
225                 //query the db for the password associated with the account
226                 db.getpassword(userInfo.username, function(err, correctPassword) {
227                     //password provided by client matches password in databse
228                     if (userInfo.password === correctPassword) {
229                         //TEMPORARY. Generate random hash to be the 'token' for this user session
230                         var randStr = Math.random().toString();
231                         var token = crypto.createHash('md5').update(randStr).digest('hex');
232                         //add user to list of active sessions
233                         activeUsers[token] = userInfo.username;
234 
235 
236                         console.log('user login. generated token: ' + token +
237                             ' for user: ' + userInfo.username);
238                         numActiveUsers++;
239                         console.log('total logged-in users: ' + numActiveUsers);
240 
241                         //give client the token
242                         socket.emit('login success', token);
243                         //redirect client to chat page
244                         socket.emit('page load');
245 
246                         //notify user's friends that they are now online
247                         for(var key in activeSockets){
248                           if(activeSockets[key].username != null){
249                             db.getfriends(activeSockets[key].username, function(err, friendList){
250                               for(var friendName in friendList){
251                                 if(friendName === userInfo.username){
252                                   io.to(key).emit('friend online', friendName);
253                                   break;
254                                 }
255                               }
256                             });
257                           }
258                         }
259                       }
260                     //client entered incorrect password
261                     else {
262                         //login failed
263                         socket.emit('login fail');
264                         console.log('login failed...\n' + 'user: ' + userInfo.username +
265                             ' pass: ' + userInfo.password);
266                     }
267                 });
268             }
269             //username does not exist in the database
270             else {
271                 //login failed because username does not exist in the database
272                 socket.emit('login fail');
273                 console.log('login failed...\n' + 'user: ' + userInfo.username +
274                     ' pass: ' + userInfo.password);
275             }
276         });
277 
278     });

280     /** used when a user requests to be logged out */
281     socket.on('logout', function(token) {
282         console.log(activeUsers);
283         console.log('attempting to logout ' + token);
284         for (var key in activeUsers) {
285             //if user is in list of active users...
286             if (key == token) {
287                 //remove user from list of active users
288                 delete activeUsers[token];
289                 //set the socket status to not-authenticated
290                 activeSockets[socketId].authenticated = false;
291                 console.log(token + ' logged out');
292                 socket.emit('logout success');
293                 break;
294             }
295         }
296     });

298     /** used when a user requests to create a new chatroom */
299     socket.on('room create request', function(roomInfo) {
300         //roomInfo.chatReceivers - list of usernames who are to be included in room
301         //roomInfo.isPrivate - true/false whether the room should be set to private
302         var chatCreator = activeSockets[socketId].username;
303         console.log(roomInfo);
304         roomInfo.chatReceivers.push(chatCreator);
305         roomInfo.chatReceivers.sort();
306 
307         //Generate hash of receiver names to act as the unique ID
308         var strToHash = "";
309         for (var key in roomInfo.chatReceivers) {
310             strToHash += roomInfo.chatReceivers[key];
311         }
312         var chatRoomId = crypto.createHash('md5').update(strToHash).digest('hex');
313 
314         //user who is creating the chatroom
315 
316         //tell the database to create a new chatroom with these users
317         db.createroom(chatRoomId, roomInfo.chatReceivers, chatCreator, roomInfo.isPrivate,
318             function(err, result) {
319                 //tell the client the room has been created
320                 socket.emit('room create success', chatRoomId);
321             });
322     });
323 
324     /** called when a user requests the list of all registered usernames */
325     socket.on('user list request', function() {
326         //get userlist from database
327         db.listusers('mainroom', function(err, usersInDb) {
328             //send userlist to the client
329             socket.emit('user list deliver', usersInDb);
330         });
331     });
332 
333     /** called when socket is disconnected */
334     socket.on('disconnect', function() {
335         //remove socket from list of connected sockets
336         delete activeSockets[socketId];
337 
338         numActiveSockets--;
339         console.log('socket disconnected: ' + clientIp);
340         console.log('total connected sockets: ' + numActiveSockets);
341     });
